  import crypto from 'crypto'

  class EncryptionService {
    private key = process.env.ENCRYPTION_KEY

    encrypt(text: string): string {
      const iv = crypto.randomBytes(16)
      const cipher = crypto.createCipher('aes-256-cbc', this.key, iv)
      let encrypted = cipher.update(text, 'utf8', 'hex')
      encrypted += cipher.final('hex')
      return iv.toString('hex') + ':' + encrypted

      const iv = Buffer.from(parts[0], 'hex')
      const encryptedText = parts[1]
      const decipher = crypto.createDecipher('aes-256-cbc', this.key, iv)
      let decrypted = decipher.update(encryptedText, 'hex', 'utf8')
      decrypted += decipher.final('utf8')
      return decrypted
    }
  }

    private analyzeCategoryPreference(tasks: Task[]) {
      const categoryCount = tasks.reduce((acc, task) => {
        acc[task.category] = (acc[task.category] || 0) + 1
        return acc
      }, {})

      return Object.entries(categoryCount)
        .sort(([,a], [,b]) => b - a)

        .map(([category, count]) => ({ category, score: count }))
    }

  // 1.2 任务推荐引擎
  class TaskRecommendationEngine {
    async getRecommendations(userId: string, limit = 5) {
      const userPreferences = await
  this.behaviorAnalyzer.analyzeTaskPreferences(userId)
      const availableTasks = await this.getAvailableTaskTemplates(userId)

      const scoredTasks = availableTasks.map(task => ({
        ...task,
        score: this.calculateRecommendationScore(task, userPreferences)
      }))

      return scoredTasks
        .sort((a, b) => b.score - a.score)
        .slice(0, limit)
    }

    private calculateRecommendationScore(task: Task, preferences: UserPreferences)    
   {
      let score = 0

      // 分类偏好权重
      const categoryPref = preferences.categories.find(c => c.category ===
  task.category)
      score += categoryPref ? categoryPref.score * 0.4 : 0

      // 难度匹配权重
      const difficultyMatch = preferences.difficulty === task.difficulty ? 0.3 : 0    
      score += difficultyMatch

      // 时长匹配权重
      const durationMatch = Math.abs(preferences.avgDuration -
  task.estimatedDuration) < 15 ? 0.2 : 0
      score += durationMatch

      // 新鲜度权重（避免重复推荐）
      const freshnessScore = this.calculateFreshnessScore(task, userId) * 0.1
      score += freshnessScore

      return score
    }

      // 分析最佳表现时间
      const hourlyPerformance = user.taskHistory.reduce((acc, task) => {
        const hour = new Date(task.completedAt).getHours()
        acc[hour] = (acc[hour] || 0) + 1
        return acc
      }, {})

      const bestHour = Object.entries(hourlyPerformance)
        .sort(([,a], [,b]) => b - a)[0]?.[0]

      return {
        bestPerformanceTime: bestHour ? `${bestHour}:00` : null,
        streakRisk: await this.calculateStreakRisk(user.userId),
        riskCategory: await this.identifyRiskCategory(user.userId)
      }
    }
  }
